```cpp
#include <string>
#include <vector>
#include <map>
#include <algorithm>
#include <iostream>
using namespace std;

vector<int> solution(vector<int> numbers) {
  vector<int> answer;
  vector<int> sums(201, 0);
  for(int i = 0; i < numbers.size() - 1; i++)
  {
    for(int j = i + 1; j < numbers.size(); j++)
    {
      int sum = numbers[i] + numbers[j];
      if(sums[sum] == 0)
      {
        answer.push_back(sum);
        sums[sum] = 1;
      }
    }
  }
  sort(answer.begin(), answer.end());
  return answer;
}
```
